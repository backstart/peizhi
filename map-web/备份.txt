#!/bin/bash
set -e
IMG="crpi-sw0t0esja4aokp42.cn-guangzhou.personal.cr.aliyuncs.com/fuyaox/fuyao-map-web:${BUILD_NUMBER}"
echo "镜像: ${IMG}"
echo "${DOCKER_PASSWORD}" | docker login --username "${DOCKER_USERNAME}" --password-stdin crpi-sw0t0esja4aokp42.cn-guangzhou.personal.cr.aliyuncs.com
docker pull "${IMG}"
docker rm -f fuyao-map-web 2>/dev/null || true
docker run -d --name fuyao-map-web --restart unless-stopped \
  -p 8002:8002 \
  -v /www/docker/fuyaomap/tiles:/data/tiles \
  -v /www/docker/fuyaomap/map-resources:/usr/share/nginx/html/map-resources \
  -v /www/docker/fuyaomapweb/runtime:/usr/share/nginx/html/runtime \
  "${IMG}"
sleep 5
curl -fsS http://127.0.0.1:8002 >/dev/null && echo "✅健康检查通过" || echo "⚠️健康检查失败，请手动验证"
docker image prune -f --filter "until=24h"
